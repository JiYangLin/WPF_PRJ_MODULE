    class MyModel : ViewModelBase
    {
        public RelayCommand MyCommand => new RelayCommand(fun);
        int val = 0;
        private void fun()
        {
            TextVal = val.ToString();
            val++;
        }

        private string _TextVal = "abcde";
        public string TextVal
        {
            get { return _TextVal; }
            set
            {
                _TextVal = value;
                RaisePropertyChanged("TextVal");
            }
        }

    }
